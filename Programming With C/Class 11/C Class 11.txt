1-10 Factorial
==================================================================

#include <stdio.h>

int main(){
    int i,factorial,j;
    for(i=1;i<=10;i++){
            factorial=1;
        for(j=1; j<=i; j++){
            factorial = factorial * j;
        }
        printf("Factorial of %d is %d\n",i,factorial);
    }

}
