for loop
--------------------------------------------
#include <stdio.h>

int main(){
    int n;
    for(n=1; n<=100; n++)
    {
        if(n%5==0){
            continue;
        }
        printf("%d\n",n);

        if(n==45){
            break;
        }
    }

    return 0;
}

while loop
---------------------------------------------------------------
#include <stdio.h>

int main(){
    int n;
    n=1;
    while(n<=100){
        printf("%d\n",n);
        n++;
    }

    return 0;
}
